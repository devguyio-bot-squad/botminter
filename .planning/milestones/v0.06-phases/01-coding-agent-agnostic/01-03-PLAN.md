The approach: Use a `.profiles_version` marker file in the profiles directory. The marker contains a version string derived from the embedded profiles' content hash. When ensure_profiles_initialized finds profiles on disk but the marker is missing or doesn't match, it re-extracts.

1. Add a constant for the version marker filename:
```rust
const PROFILES_VERSION_MARKER: &str = ".profiles_version";
```

2. Add a function to compute the embedded profiles version. Use a simple approach — hash the concatenation of all embedded profile names + the binary's compile-time env. The simplest reliable approach: use `env!("CARGO_PKG_VERSION")` combined with a compile-time constant that changes when profiles change. Since profiles are embedded via include_dir, the binary is recompiled when they change. So `env!("CARGO_PKG_VERSION")` plus a manually bumped const is sufficient. Even simpler: just use a hardcoded `PROFILES_FORMAT_VERSION` constant that developers bump when profile structure changes:

```rust
/// Bump this when the profile directory structure changes in ways that
/// require re-extraction (e.g., members/ -> roles/ rename).
const PROFILES_FORMAT_VERSION: &str = "2";
```

3. Modify `extract_embedded_to_disk` to write the marker file after successful extraction:
```rust
// At the end of extract_embedded_to_disk, after the loop:
let marker_path = target.join(PROFILES_VERSION_MARKER);
fs::write(&marker_path, PROFILES_FORMAT_VERSION)?;
```

4. Modify `ensure_profiles_initialized_with` to check the marker when profiles exist:
```rust
fn ensure_profiles_initialized_with(...) -> Result<()> {
    if profiles_path.is_dir() {
        let has_profile = fs::read_dir(profiles_path)
            .map(|entries| entries.filter_map(|e| e.ok())
                .any(|e| e.file_type().map(|t| t.is_dir()).unwrap_or(false)))
            .unwrap_or(false);
        if has_profile {
            // Check version marker — re-extract if stale
            let marker_path = profiles_path.join(PROFILES_VERSION_MARKER);
            let current_version = fs::read_to_string(&marker_path).unwrap_or_default();
            if current_version.trim() == PROFILES_FORMAT_VERSION {
                return Ok(());
            }
            // Stale — re-extract silently
            eprintln!("Profiles outdated (format v{} -> v{}), re-extracting...",
                current_version.trim(), PROFILES_FORMAT_VERSION);
            let count = embedded::extract_embedded_to_disk(profiles_path)?;
            eprintln!("Updated {} profiles in {}", count, profiles_path.display());
            return Ok(());
        }
    }
    // ... rest unchanged (TTY prompt / auto-init for missing profiles)
}
```

#[test]
fn ensure_profiles_initialized_re_extracts_when_marker_missing() {
    // Extract profiles, then delete the marker
    let tmp = tempfile::tempdir().unwrap();
    let profiles_path = tmp.path().join("profiles");
    fs::create_dir_all(&profiles_path).unwrap();
    embedded::extract_embedded_to_disk(&profiles_path).unwrap();

    // Remove marker to simulate stale profiles
    fs::remove_file(profiles_path.join(PROFILES_VERSION_MARKER)).unwrap();

    // Simulate stale layout by renaming roles/ to members/
    for name in embedded::list_embedded_profiles() {
        let roles = profiles_path.join(&name).join("roles");
        let members = profiles_path.join(&name).join("members");
        if roles.exists() {
            fs::rename(&roles, &members).unwrap();
        }
    }

    // ensure_profiles_initialized should re-extract
    let mut reader = io::Cursor::new(b"");
    ensure_profiles_initialized_with(&profiles_path, false, &mut reader).unwrap();

    // Verify roles/ exists again (re-extracted)
    for name in embedded::list_embedded_profiles() {
        let roles = profiles_path.join(&name).join("roles");
        assert!(roles.is_dir(), "roles/ should exist after re-extraction for {}", name);
    }
}

#[test]
fn ensure_profiles_initialized_skips_when_marker_current() {

    let profiles_path = tmp.path().join("profiles");

    // Verify marker still matches

    assert!(marker_path.exists(), ".profiles_version marker should exist");
    let content = fs::read_to_string(&marker_path).unwrap();
    assert_eq!(content.trim(), PROFILES_FORMAT_VERSION);
}
