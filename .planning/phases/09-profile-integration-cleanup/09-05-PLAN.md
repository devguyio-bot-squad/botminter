let bridges: Vec<String> = {

    let mut names = Vec::new();
    if let Ok(profile_names) = profile::list_profiles() {
        for pn in &profile_names {
            if let Ok(m) = profile::read_manifest(pn) {
                for b in &m.bridges {
                    if !names.contains(&b.name) {
                        names.push(b.name.clone());
                    }
                }
            }
        }
