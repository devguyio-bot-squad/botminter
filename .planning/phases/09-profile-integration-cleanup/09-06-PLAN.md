**B) Pre-existence check in create_workspace_repo (workspace.rs):**

Before the `gh repo create` call (line 68-70), add a check using `gh repo view`:

```rust
// Check if repo already exists on GitHub
let mut view_cmd = Command::new("gh");
view_cmd.args(["repo", "view", &ws_repo_name, "--json", "name"]);
if let Some(token) = params.gh_token {
    view_cmd.env("GH_TOKEN", token);
}
let view_output = view_cmd.output().context("Failed to run `gh repo view`")?;
if view_output.status.success() {
    // Repo already exists — skip creation, just clone it
    log::info!("Workspace repo '{}' already exists on GitHub, skipping creation.", ws_repo_name);
} else {
    // Create GitHub repo
    let mut cmd = Command::new("gh");
    cmd.args(["repo", "create", &ws_repo_name, "--private"]);
    if let Some(token) = params.gh_token {
        cmd.env("GH_TOKEN", token);
    }
    let output = cmd.output().context("Failed to run `gh repo create`")?;
    if !output.status.success() {
        let stderr = String::from_utf8_lossy(&output.stderr);
        bail!(
            "Failed to create workspace repo '{}'.\n{}\n\n\
             If the repo already exists:\n  \
             gh repo delete {} --yes\n\
             Then re-run `bm teams sync --repos`.",
            ws_repo_name,
            stderr.trim(),
            ws_repo_name,
        );
    }
}
```

Note: `params.gh_token` is used in both the view and create calls. Since it's an `Option<&str>`, the borrow checker may require adjusting — if `params.gh_token` is consumed, clone it before the first use.
