Add bridge auto-start logic BEFORE member launch (after team resolution, after manifest loading):

```rust
// Bridge auto-start (before members)
if !no_bridge {
    if let Some(bridge_dir) = bridge::discover(&team_repo, &team.name)? {
        let manifest = bridge::load_manifest(&bridge_dir)?;
        if manifest.spec.bridge_type == "local" {
            if let Some(lifecycle) = &manifest.spec.lifecycle {
                println!("Starting bridge '{}'...", manifest.metadata.name);
                // Check if just is available
                if which::which("just").is_err() {
                    eprintln!("Warning: 'just' not found. Skipping bridge start. Install: https://just.systems/");
                } else {
                    bridge::invoke_recipe(&bridge_dir, &lifecycle.start, &[], &team.name)?;
                    bridge::invoke_recipe(&bridge_dir, &lifecycle.health, &[], &team.name)?;

                    let state_path = bridge::state_path(&cfg.workzone, &team.name);
                    let mut state = bridge::load_state(&state_path)?;
                    state.bridge_name = Some(manifest.metadata.name.clone());
                    state.bridge_type = Some(manifest.spec.bridge_type.clone());
                    state.status = "running".to_string();
                    state.started_at = Some(chrono::Utc::now().to_rfc3339());
                    state.last_health_check = Some(chrono::Utc::now().to_rfc3339());
                    // Read service URL from config exchange if available
                    // (already handled by invoke_recipe return value)
                    bridge::save_state(&state_path, &state)?;
                    println!("Bridge '{}' started.", manifest.metadata.name);
                }
            }
        }
        // External bridges: no lifecycle to start, but note it
        if manifest.spec.bridge_type == "external" {
            println!("Bridge '{}' is external (managed externally).", manifest.metadata.name);
        }
    }
}

if bridge_only {
    return Ok(());
}
// ... existing member launch code follows ...
```

If `bridge_only` is true, return Ok(()) after bridge start (skip member launch). If `no_bridge` is true, skip the bridge block entirely. If no bridge is configured, silently continue (no error).

// Stop bridge if configured and running
if let Some(bridge_dir) = bridge::discover(&team_repo, &team.name)? {
    let manifest = bridge::load_manifest(&bridge_dir)?;
    if manifest.spec.bridge_type == "local" {
        if let Some(lifecycle) = &manifest.spec.lifecycle {
            if which::which("just").is_ok() {
                let state_path = bridge::state_path(&cfg.workzone, &team.name);
                let state = bridge::load_state(&state_path)?;
                if state.status == "running" {
                    println!("Stopping bridge '{}'...", manifest.metadata.name);
                    bridge::invoke_recipe(&bridge_dir, &lifecycle.stop, &[], &team.name)?;
                    let mut state = state;
                    state.status = "stopped".to_string();
                    bridge::save_state(&state_path, &state)?;
                    println!("Bridge '{}' stopped.", manifest.metadata.name);
                }
            }
        }
    }
}
